<script>
    function createMiniChart(ctx, data) {
        new Chart(ctx, {
            type: 'line',
            data: {
                labels: Array(data.length).fill(''),
                datasets: [{
                    data: data,
                    borderColor: '#ffffff',
                    backgroundColor: 'rgba(255,255,255,0.3)',
                    borderWidth: 2,
                    pointRadius: 0,
                    tension: 0.4,
                    fill: true
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { display: false },
                    y: { display: false }
                },
                elements: { line: { tension: 0.4 } },
                animation: { duration: 1000, easing: 'easeOutQuart' }
            }
        });
    }

    window.addEventListener('load', () => {
        createMiniChart(document.getElementById('chart1'), [5,6,7,9,9,5,3,2,4,6,8]);
        createMiniChart(document.getElementById('chart2'), [2,3,5,8,7,6,9,4,3,7,8]);
        createMiniChart(document.getElementById('chart3'), [3,4,6,5,9,8,7,6,5,7,9]);
        createMiniChart(document.getElementById('chart4'), [9,7,6,8,9,5,4,3,7,8,9]);
    });
</script>
