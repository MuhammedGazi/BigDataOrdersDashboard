<div class="mt-3 text-center">
    <nav aria-label="Page navigation">
        <ul class="pagination justify-content-center">

            @if (ViewBag.CurrentPage > 1)
            {
                <li class="page-item">
                    <a class="page-link" href="/Product/ProductList?page=@(ViewBag.CurrentPage - 1)">Önceki</a>
                </li>
            }

            @{
                int startPage = Math.Max(1, ViewBag.CurrentPage - 2);
                int endPage = Math.Min(ViewBag.TotalPages, ViewBag.CurrentPage + 2);

                if (startPage > 1)
                {
                    <li class="page-item">
                        <a class="page-link" href="/Product/ProductList?page=1">1</a>
                    </li>
                    if (startPage > 2)
                    {
                        <li class="page-item disabled">
                            <span class="page-link">...</span>
                        </li>
                    }
                }

                for (int i = startPage; i <= endPage; i++)
                {
                    <li class="page-item @(i == ViewBag.CurrentPage ? "active" : "")">
                        <a class="page-link" href="/Product/ProductList?page=@i">@i</a>
                    </li>
                }

                if (endPage < ViewBag.TotalPages)
                {
                    if (endPage < ViewBag.TotalPages - 1)
                    {
                        <li class="page-item disabled">
                            <span class="page-link">...</span>
                        </li>
                    }
                    <li class="page-item">
                        <a class="page-link" href="/Product/ProductList?page=@ViewBag.TotalPages">@ViewBag.TotalPages</a>
                    </li>
                }
            }

            @if (ViewBag.CurrentPage < ViewBag.TotalPages)
            {
                <li class="page-item">
                    <a class="page-link" href="/Product/ProductList?page=@(ViewBag.CurrentPage + 1)">Sonraki</a>
                </li>
            }

        </ul>
    </nav>
</div>
